**init.sql**
```sql
-- Enable pgvector extension
CREATE EXTENSION IF NOT EXISTS vector;

-- Nodes table (discrete units of meaning)
CREATE TABLE nodes (
    node_id SERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    embedding VECTOR(384), -- sentence-transformers dimension
    frequency INTEGER DEFAULT 1,
    weight FLOAT DEFAULT 1.0,
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW()
);

-- Topics table (emergent from weighted node convergence)
CREATE TABLE topics (
    topic_id SERIAL PRIMARY KEY,
    label TEXT NOT NULL,
    centroid_embedding VECTOR(384),
    origin_node_ids INTEGER[],
    total_weight FLOAT,
    coherence_score FLOAT,
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW()
);

-- Documents table
CREATE TABLE documents (
    doc_id SERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    embedding VECTOR(384),
    topic_ids INTEGER[],
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW()
);

-- Node relationships for topic emergence
CREATE TABLE node_relationships (
    rel_id SERIAL PRIMARY KEY,
    node_id_1 INTEGER REFERENCES nodes(node_id),
    node_id_2 INTEGER REFERENCES nodes(node_id),
    similarity_score FLOAT,
    co_occurrence_count INTEGER DEFAULT 1,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Indexes for fast similarity search
CREATE INDEX ON nodes USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100);
CREATE INDEX ON topics USING ivfflat (centroid_embedding vector_cosine_ops) WITH (lists = 100);
CREATE INDEX ON documents USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100);
```

**rag_system.py**
```python
import os
import numpy as np
import pandas as pd
from typing import List, Dict, Any, Tuple, Optional
from sentence_transformers import SentenceTransformer
from sqlalchemy import create_engine, text
from sqlalchemy.orm import sessionmaker
from sklearn.cluster import DBSCAN
from sklearn.metrics.pairwise import cosine_similarity
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class RAGTopicSystem:
    def __init__(self, db_url: str = None):
        self.db_url = db_url or "postgresql://rag_user:rag_pass@localhost:5432/rag_system"
        self.engine = create_engine(self.db_url)
        self.Session = sessionmaker(bind=self.engine)
        
        # Load sentence transformer for embeddings
        self.embedder = SentenceTransformer('all-MiniLM-L6-v2')
        
        # Topic emergence thresholds
        self.convergence_threshold = 0.7
        self.min_cluster_size = 3
        self.coherence_threshold = 0.6
        
    def embed_text(self, text: str) -> np.ndarray:
        """Generate embedding for text"""
        return self.embedder.encode(text)
    
    def insert_node(self, content: str, metadata: Dict = None) -> int:
        """Insert a new node and return its ID"""
        embedding = self.embed_text(content)
        metadata = metadata or {}
        
        with self.Session() as session:
            result = session.execute(text("""
                INSERT INTO nodes (content, embedding, metadata)
                VALUES (:content, :embedding, :metadata)
                RETURNING node_id
            """), {
                'content': content,
                'embedding': embedding.tolist(),
                'metadata': metadata
            })
            node_id = result.fetchone()[0]
            session.commit()
            logger.info(f"Inserted node {node_id}: {content[:50]}...")
            return node_id
    
    def find_similar_nodes(self, embedding: np.ndarray, limit: int = 10, 
                          threshold: float = 0.7) -> List[Dict]:
        """Find nodes similar to given embedding"""
        with self.Session() as session:
            result = session.execute(text("""
                SELECT node_id, content, weight, 
                       (embedding <=> :embedding) as distance,
                       (1 - (embedding <=> :embedding)) as similarity
                FROM nodes
                WHERE (1 - (embedding <=> :embedding)) > :threshold
                ORDER BY embedding <=> :embedding
                LIMIT :limit
            """), {
                'embedding': embedding.tolist(),
                'threshold': threshold,
                'limit': limit
            })
            return [dict(row._mapping) for row in result.fetchall()]
    
    def update_node_relationships(self, node_id: int):
        """Update relationships for a node with similar nodes"""
        with self.Session() as session:
            # Get the node's embedding
            result = session.execute(text("""
                SELECT embedding FROM nodes WHERE node_id = :node_id
            """), {'node_id': node_id})
            
            node_embedding = np.array(result.fetchone()[0])
            similar_nodes = self.find_similar_nodes(node_embedding, limit=20)
            
            # Insert/update relationships
            for similar_node in similar_nodes:
                if similar_node['node_id'] != node_id:
                    session.execute(text("""
                        INSERT INTO node_relationships 
                        (node_id_1, node_id_2, similarity_score, co_occurrence_count)
                        VALUES (:node_id_1, :node_id_2, :similarity, 1)
                        ON CONFLICT (node_id_1, node_id_2) 
                        DO UPDATE SET 
                            co_occurrence_count = node_relationships.co_occurrence_count + 1,
                            similarity_score = :similarity
                    """), {
                        'node_id_1': min(node_id, similar_node['node_id']),
                        'node_id_2': max(node_id, similar_node['node_id']),
                        'similarity': similar_node['similarity']
                    })
            session.commit()
    
    def detect_emerging_topics(self) -> List[Dict]:
        """Detect topics from node convergence"""
        with self.Session() as session:
            # Get all nodes with their embeddings
            result = session.execute(text("""
                SELECT node_id, content, embedding, weight
                FROM nodes
                ORDER BY weight DESC
            """))
            nodes_data = [dict(row._mapping) for row in result.fetchall()]
        
        if len(nodes_data) < self.min_cluster_size:
            return []
        
        # Prepare embeddings for clustering
        embeddings = np.array([node['embedding'] for node in nodes_data])
        node_ids = [node['node_id'] for node in nodes_data]
        
        # Perform clustering
        clustering = DBSCAN(
            eps=1-self.convergence_threshold,  # Convert similarity to distance
            min_samples=self.min_cluster_size,
            metric='cosine'
        ).fit(embeddings)
        
        emerging_topics = []
        
        for cluster_id in set(clustering.labels_):
            if cluster_id == -1:  # Skip noise
                continue
                
            # Get nodes in this cluster
            cluster_mask = clustering.labels_ == cluster_id
            cluster_nodes = [nodes_data[i] for i in np.where(cluster_mask)[0]]
            cluster_embeddings = embeddings[cluster_mask]
            
            # Calculate coherence score
            if len(cluster_embeddings) > 1:
                coherence = np.mean(cosine_similarity(cluster_embeddings))
            else:
                coherence = 1.0
            
            if coherence >= self.coherence_threshold:
                # Create topic
                centroid = np.mean(cluster_embeddings, axis=0)
                origin_node_ids = [node['node_id'] for node in cluster_nodes]
                total_weight = sum(node['weight'] for node in cluster_nodes)
                
                # Generate topic label (using most weighted node's content)
                top_node = max(cluster_nodes, key=lambda x: x['weight'])
                label = f"Topic_{cluster_id}_{top_node['content'][:30]}"
                
                emerging_topics.append({
                    'label': label,
                    'centroid_embedding': centroid,
                    'origin_node_ids': origin_node_ids,
                    'total_weight': total_weight,
                    'coherence_score': coherence,
                    'metadata': {'cluster_size': len(cluster_nodes)}
                })
        
        return emerging_topics
    
    def save_topics(self, topics: List[Dict]) -> List[int]:
        """Save emerging topics to database"""
        topic_ids = []
        
        with self.Session() as session:
            for topic in topics:
                result = session.execute(text("""
                    INSERT INTO topics 
                    (label, centroid_embedding, origin_node_ids, total_weight, 
                     coherence_score, metadata)
                    VALUES (:label, :centroid_embedding, :origin_node_ids, 
                            :total_weight, :coherence_score, :metadata)
                    RETURNING topic_id
                """), {
                    'label': topic['label'],
                    'centroid_embedding': topic['centroid_embedding'].tolist(),
                    'origin_node_ids': topic['origin_node_ids'],
                    'total_weight': topic['total_weight'],
                    'coherence_score': topic['coherence_score'],
                    'metadata': topic['metadata']
                })
                topic_id = result.fetchone()[0]
                topic_ids.append(topic_id)
                logger.info(f"Saved topic {topic_id}: {topic['label']}")
            
            session.commit()
        
        return topic_ids
    
    def classify_input(self, text: str) -> Dict[str, Any]:
        """Classify input text using RAG system"""
        embedding = self.embed_text(text)
        
        with self.Session() as session:
            # Find similar topics
            topics_result = session.execute(text("""
                SELECT topic_id, label, total_weight, coherence_score,
                       (1 - (centroid_embedding <=> :embedding)) as similarity
                FROM topics
                ORDER BY centroid_embedding <=> :embedding
                LIMIT 5
            """), {'embedding': embedding.tolist()})
            
            similar_topics = [dict(row._mapping) for row in topics_result.fetchall()]
            
            # Find similar nodes
            similar_nodes = self.find_similar_nodes(embedding, limit=10)
            
            # Calculate classification confidence
            if similar_topics:
                best_topic = similar_topics[0]
                confidence = best_topic['similarity'] * best_topic['coherence_score']
            else:
                best_topic = None
                confidence = 0.0
        
        return {
            'input_text': text,
            'best_topic': best_topic,
            'similar_topics': similar_topics,
            'similar_nodes': similar_nodes,
            'confidence': confidence,
            'embedding': embedding.tolist()
        }
    
    def process_document(self, content: str, metadata: Dict = None) -> Dict:
        """Process a document: extract nodes, detect topics, classify"""
        # Simple sentence splitting for node extraction
        sentences = [s.strip() for s in content.split('.') if len(s.strip()) > 10]
        
        node_ids = []
        for sentence in sentences:
            node_id = self.insert_node(sentence, metadata)
            node_ids.append(node_id)
            self.update_node_relationships(node_id)
        
        # Detect new topics
        emerging_topics = self.detect_emerging_topics()
        topic_ids = self.save_topics(emerging_topics) if emerging_topics else []
        
        # Insert document
        doc_embedding = self.embed_text(content)
        with self.Session() as session:
            result = session.execute(text("""
                INSERT INTO documents (content, embedding, topic_ids, metadata)
                VALUES (:content, :embedding, :topic_ids, :metadata)
                RETURNING doc_id
            """), {
                'content': content,
                'embedding': doc_embedding.tolist(),
                'topic_ids': topic_ids,
                'metadata': metadata or {}
            })
            doc_id = result.fetchone()[0]
            session.commit()
        
        return {
            'doc_id': doc_id,
            'node_ids': node_ids,
            'topic_ids': topic_ids,
            'emerging_topics_count': len(emerging_topics)
        }
```

from rag_system import RAGTopicSystem
import time

def main():

                SELECT label, total_weight, coherence_score 
                FROM topics 
                ORDER BY total_weight DESC 
                LIMIT 5
            """))
            
            print(f"\n🏆 Top Topics:")
            for topic in result.fetchall():

## 🎯 Key Features Implemented

1. **Node-based Architecture**: Each meaningful unit gets stored as a node with embeddings
2. **Topic Emergence**: DBSCAN clustering detects topics from weighted node convergence  
3. **Fast Vector Search**: pgvector with IVFFLAT indexes for millisecond similarity queries
4. **RAG Classification**: Input queries get matched against both topics and nodes
5. **Relationship Tracking**: Node co-occurrence and similarity relationships
6. **Scalable Design**: PostgreSQL handles millions of nodes/topics efficiently
